#include<bits/stdc++.h>

using namespace std;

#define int long long

signed main() {
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int t;
    cin >> t;
    while(t --){
        int n;
        cin >> n;
        if(n % 2 == 0) {
            for(int i=n/2;i<=n-1;i++) cout << i << ' ';
            for(int i=n+1;i<=3*n/2;i++) cout << i << ' ';
        }
        else {
            vector<int> ans(n);
            iota(ans.begin(), ans.end(), (n + 5) / 2);
            ans[0]--;
            ans[n - 1]++;
            ans[n - 2]++;
